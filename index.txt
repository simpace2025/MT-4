  <script>
    /*
      Mỗi lần upload ảnh/audio mới lên GitHub, đổi VERSION:
      Ví dụ: 20260514_01 -> 20260514_02 -> 20260514_03
    */
    const VERSION = "20260514_01";

    function withVersion(path){
      return path + "?v=" + VERSION;
    }

    // Tự thử nhiều đuôi ảnh: .jpg, .JPG, .png, .PNG
    function loadSmartImage(img){
      const base = img.getAttribute("data-img");
      const candidates = [
        base + ".jpg",
        base + ".JPG",
        base + ".jpeg",
        base + ".JPEG",
        base + ".png",
        base + ".PNG"
      ];

      let index = 0;

      function tryNext(){
        if(index >= candidates.length){
          const warn = document.createElement("div");
          warn.className = "missing";
          warn.textContent = "Không tìm thấy ảnh: " + base + " (.jpg/.JPG/.png). Kiểm tra lại tên file trên GitHub.";
          img.replaceWith(warn);
          return;
        }

        const test = new Image();
        test.onload = () => {
          img.src = withVersion(candidates[index]);
        };
        test.onerror = () => {
          index++;
          tryNext();
        };
        test.src = withVersion(candidates[index]);
      }

      tryNext();
    }

    // Audio: thử nhiều đường dẫn phổ biến để tránh lỗi chữ hoa/thường
    function setupAudio(audio){
      const sources = JSON.parse(audio.getAttribute("data-audio"));
      let index = 0;

      function tryAudio(){
        if(index >= sources.length){
          const warn = document.createElement("div");
          warn.className = "missing";
          warn.textContent = "Không tìm thấy audio. Hãy kiểm tra tên file trong folder AUDIO/Audio.";
          audio.replaceWith(warn);
          return;
        }

        audio.src = withVersion(sources[index]);
        audio.load();
      }

      audio.addEventListener("error", () => {
        index++;
        tryAudio();
      });

      tryAudio();
    }

    function addImages(containerId, prefix, start, end){
      const container = document.getElementById(containerId);
      for(let i = start; i <= end; i++){
        const img = document.createElement("img");
        img.className = "test-img";
        img.setAttribute("data-img", "JPG1/" + prefix + i);
        img.alt = prefix + i;
        container.appendChild(img);
      }
    }

    window.addEventListener("DOMContentLoaded", () => {
      // Logo nằm cùng thư mục với index.html. Đặt tên file logo là logo.png
      const logo = document.querySelector(".logo");
      if(logo){
        logo.src = withVersion("logo.jpg");
      }

      // Render Reading & Writing: RW1 -> RW16
      addImages("rw-images", "RW", 1, 16);

      // Render Speaking: S1 -> S6
      addImages("speaking-images", "S", 1, 6);

      // Load ảnh thông minh
      document.querySelectorAll("img[data-img]").forEach(loadSmartImage);

      // Load audio
      document.querySelectorAll("audio[data-audio]").forEach(setupAudio);

      // Cập nhật ?v= trên thanh địa chỉ để tránh cache HTML
      const url = new URL(window.location.href);
      if (url.searchParams.get("v") !== VERSION) {
        url.searchParams.set("v", VERSION);
        history.replaceState(null, "", url.toString());
      }
    });
  </script>
